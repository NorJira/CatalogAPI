// command.txt
// 
// add MongoDb Client package
%> dotnet add package MongoDB.Driver 

%> dotnet user-secrets init 

%> dotnet user-secrets set MongoDbSettings:Password pass#word1 

// pull mongodb image and create in docker container with volume to persist data
%> docker run -d --rm --name mongo -p 27017:27017 -v mongodbdata:/data/db mongo
// OR
%> docker run -d --rm --name mongo -p 27017:27017 -v mongodbdata:/data/db -e MONGO_INITDB_ROOT_USERNAME=mongoadmin -e MONGO_INITDB_ROOT_PASSWORD=pass#word1 mongo dock

// show docker container running
%> docker ps 

// show docker volume
%> docker volume ls 

// remove docker volume
%> docker volume rm mongo
